算法：封装一个函数，功能如下,输入英文字符串，返回其中出现次数最多的字母以及其出现的次数。
function fn(str) {
    var max = 0;
    var result = "";
    for (var i = 0; i < str.length; i++) {
        var len = str.split(str.charAt(i)).length - 1;
        if (max < len) {
            max = len;
            result = str.charAt(i)
        } else if (max === len && result.indexOf(str.charAt(i)) === -1) {
            result += str.charAt(i)
        }
    }
    return result + "出现的次数最多：" + max + "次";
}
